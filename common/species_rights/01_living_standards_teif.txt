living_standard_hive_mind = {
	token = living_standard_hive_mind
	name = living_standard_hive_mind

	potential = {
		OR = {
			from = { has_ethic = ethic_gestalt_consciousness }
			AND = {
				from = { has_country_flag = synthetic_empire }
				has_trait = trait_machine_unit
			}
		}
	}	
	
	allow = {
		hidden_trigger = {
			from = {
				OR = {
					has_ethic = ethic_gestalt_consciousness
					has_country_flag = synthetic_empire
				}
			}
		}
		custom_tooltip = {
			fail_text = species_is_not_hive
			OR = {
				AND = {
					has_trait = trait_hive_mind
					from = { has_authority = auth_hive_mind }
				}
				NOT = { from = { has_authority = auth_hive_mind } }
			}
		}
		custom_tooltip = {
			fail_text = species_is_not_machine
			OR = {
				NOT = { from = { has_authority = auth_machine_intelligence } }
				AND = {
					from = { has_authority = auth_machine_intelligence }
					OR = {
						has_trait = trait_machine_unit
						has_trait = trait_mechanical
						AND = {
							from = { has_civic = civic_machine_assimilator }
							has_trait = trait_cybernetic
						}
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = SPECIES_MUST_NOT_BE_ASSIMILATING
			NOT = {
				has_citizenship_type = {
					type = citizenship_assimilation
					country = from
				}
			}
		}
		custom_tooltip = {
			fail_text = species_is_not_machine
			OR = {
				NOT = { from = { has_country_flag = synthetic_empire } }
				AND = {
					from = { has_country_flag = synthetic_empire }
					has_trait = trait_machine_unit
					has_citizenship_type = {
						type = citizenship_full_machine
						country = from
					}
				}
			}
		}
	}
	
	ai_will_do = {
		factor = 1
		modifier = {
			factor = 10
			NOT = { is_same_species = from }
			from = {
				NOT = { has_ethic = ethic_gestalt_consciousness }
				has_country_flag = synthetic_empire
				is_xenophobe = no
			}
		}
	}
}
