# Declare war, usable by regular, fallen & awakened empires
action_declare_war = {
	icon = "GFX_diplomacy_status_war"
	requires_actor_independence = no # You are able to declare war only on your overlord when being a vassal, this special case is handled in code instead.
	requires_recipient_independence = no
	requires_alliance_vote = yes
	requires_unanimous_vote = yes
	auto_accepted = yes
	should_show_auto_accept_message_actor = yes
	should_show_auto_accept_message_recipient = yes
	should_open_auto_accept_message_recipient = yes
	should_notify_all_communications = yes
	action_type = aggressive
    possible = {
		hidden_trigger = {
			NOT = { has_country_flag = ai_no_wars }
		}
    }
	on_accept = {
        hidden_effect = {
            if = {
                limit = {
                    from = { has_valid_civic = civic_gestalt_terminator }
					is_gestalt = yes
                }
                from = {
                    set_war_goal = { type = wg_end_threat_gestalt target = from enemy = root }
                }
			}
			if = {
				limit = {
					any_war = {
						any_attacker = {
							is_same_value = root
						}
						any_attacker = {
							NOT = { is_same_value = root }
							is_fallen_empire = yes
							is_ai = yes
						}
					}
				}
				every_war = {
					limit = {
						any_attacker = { is_same_value = root }
						any_attacker = {
							NOT = { is_same_value = root }
							is_fallen_empire = yes
							is_ai = yes
						}
					}
					every_war_participant = {
						limit = {
							NOT = { is_same_value = root }
							is_fallen_empire = yes
							is_ai = yes
						}
						prev = { remove_war_participant = prev }
					}
				}
			}
        }
	}
}

# Send an insult. Harms relations.

	icon = "GFX_diplomacy_status_war"

	requires_actor_independence = no
	auto_accepted = yes
	should_show_auto_accept_message_recipient = no
	action_type = neutral
	possible = {
		if = {
			limit = { is_ai = yes }
			from = { NOT = { has_relation_flag = { who = root flag = timed_insult } } }
		}
		if = {
			limit = { is_ai = no }
			custom_tooltip = {
				fail_text = "already_ex_diplomacy"
				NOT = { has_country_flag = ex_diplomacy }
			}
		}
	}
	on_accept = {
        hidden_effect = {
            if = {
                limit = {
                    is_ai = no
                }
                from = { remove_opinion_modifier = { who = root modifier = opinion_insult } }
                open_more_diplomacy = yes
                break = yes
            }
            if = {
                limit = {
                    is_ai = yes
                }
                from = {
                    remove_opinion_modifier = { who = root modifier = opinion_insult }
                    set_timed_relation_flag = { flag = timed_insult who = root days = 30 }
                }
                open_more_diplomacy_ai = yes
                break = yes
            }
        }
	}
}
